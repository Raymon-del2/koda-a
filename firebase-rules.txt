rules_version = '2';
service cloud.firestore {
  match /databases/{database}/documents {
    // Feedback collection - allow anyone to create feedback
    match /feedback/{feedbackId} {
      // Allow anyone to create feedback (for anonymous users)
      allow create: if request.auth == null || request.auth != null;
      // Only allow reading feedback for authenticated users (admin purposes)
      allow read: if request.auth != null;
      // Don't allow updates or deletes for security
      allow update, delete: if false;
    }

    // Users collection - existing rules
    match /users/{userId} {
      allow read, write: if request.auth != null && request.auth.uid == userId;
    }

    // Guided learning materials - existing rules
    match /guidedLearning/{userId} {
      allow read, write: if request.auth != null && request.auth.uid == userId;
    }
  }
}
